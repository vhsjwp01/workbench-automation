#!/usr/bin/ksh
#
# This script should be run before Build to correct various
# anomalies in the code.

export PROJECT=~/workspace/GRMSSCAN

echo "Change BATCH line sequential to sequential"

cd $PROJECT/target/BATCH
for i in *.cbl
do
cp --preserve=timestamps $i $i.bak
mv $i $i.sed80
sed -e s/"ORGANIZATION LINE SEQUENTIAL"/"organization sequential"/ <$i.sed80 > $i.sed99
if diff $i.sed99 $i.bak > /dev/null ; then
  cp --preserve=timestamps $i.bak $i
  echo "Processing " $i " NO CHANGES"
else
  echo "Processing " $i " UPDATED **************"
  cp $i.sed99 $i
fi
done
